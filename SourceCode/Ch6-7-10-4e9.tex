\Anchor{4e9A4}\ProblemN{A.4}{
	\TextA{Supp $V$ is real inner prodsp, $\beta$ is biliney on $V.$ Show $\exists\,!\,T\in\Lm{V},\,\beta\Par{u,v}=\Ang{Tu,v}.$}
}Let orthon $B_V=\Par{e_1,\dots,e_n}\Rightarrow\beta\Par{u,v}=\sum\beta\Par{u,e_i}\Ang{e_i,v}=\BigAng{{\sum\beta\Par{u,e_i}e_i},\:\!v}.$\parSol{}
Supp $\beta\Par{u,v}=\Ang{Tu,v}=\sum\Ang{Tu,e_i}\Ang{e_i,v}.$ Then each $\Ang{Tu,e_i}=\beta\Par{u,e_i}\Rightarrow Tu=\sum\beta\Par{u,e_i}e_i.$\PfEnd\vspace{3pt}\parSol{}
\Or Define $Te_j=\sum\beta\Par{e_j,e_i}e_i.$ Then $\Ang{Te_j,e_k}=\beta\Par{e_j,e_k}.$\parSol{}
Thus $\Ang{Tu,v}=\sum a_jb_k\Ang{Te_j,e_k}=\sum a_jb_k\;\!\beta\Par{e_j,e_k}=\beta\Par{u,v}.$\PfEnd

\Anchor{4e9B1}\ProblemN{B.1}{
	\TextA{Supp $m\in\Nbp.$ Show $\dim V^{\SmallPar{m}}=\dim V^m.$}
}We use induc on $m.$ (i) $m=1.$ Immed. (ii) $m>1.$ Asum true for $\Par{m-1}.$\parSol{}
See (4E 9.D). Becs $\mB\BigPar{V^{\SmallPar{m-1}},V}$ is iso to $V^{\SmallPar{m}}$ via $\Sbra{\Lambda\Par{\beta}}\Par{v_1,\dots,v_m}=\beta\BigPar{\Par{v_1,\dots,v_{m-1}},v_m}.$\PfEnd\vspace{3pt}\parSol{}
Let $B_V=\Par{e_1,\dots,e_n}$ with corres $B_{V\apostrophe}=\Par{\varphi_1,\dots,\varphi_n}.$\parSol{}
For each $\Par{i_1,\dots,i_m}\in\!\!\;\Bra{1,\dots,n}{^m}=\Gamma,$ define $m$-liney \,$\alpha_{i_1,\cdots,\,i_m}\Par{v_1,\dots,v_m}=\varphi_{i_1}\:\!\!\Par{v_1}\cdots\varphi_{i_m}\:\!\!\Par{v_m}.$\vspace{1pt}\parSol{}
Supp $\alpha$ is $m$-liney. Note that $\alpha\Par{v_1,\dots,v_m}=\sum_{i_1=1}^n\cdots\sum_{i_m=1}^n\,\alpha_{i_1,\cdots,\,i_m}\Par{v_1,\dots,v_m}\,\alpha\Par{e_{i_1},\dots,e_{i_m}}.$\vspace{2pt}\parSol{}
Hence $\alpha\in\Span{\alpha_I}{_{I\,\in\,\Gamma}}.$ \;For liney indep, let $\sum_{i_1=1}^n\cdots\sum_{i_m=1}^n\,c_{i_1,\cdots,\,i_m}\,\alpha_{i_1,\cdots,\,i_m}=0.$\vspace{1pt}\parSol{}
Note that for $\Par{k_1,\dots,k_m}\in\Gamma\:\!\!,\:\alpha_{i_1,\cdots,\,i_m}\Par{e_{k_1},\dots,e_{k_m}}=\delta_{i_1,\,k_1}\!\!\cdots\delta_{i_m,\,k_m}\Rightarrow 0\Par{e_{k_1},\dots,e_{k_m}}=c_{k_1,\cdots,\,k_m}.$\PfEnd
